cmake_minimum_required(VERSION 2.8.12)
project(MagnumMultiWindowExample)

# CMake policies: enable MACOSX_RPATH by default
if(POLICY CMP0042)
    cmake_policy(SET CMP0042 NEW)
endif()
# Don't treat imported targets with :: as files
if(POLICY CMP0028)
    cmake_policy(SET CMP0028 NEW)
endif()

find_package(Magnum REQUIRED
    MeshTools
    SceneGraph
    Shaders
    Sdl2Application)

set_directory_properties(PROPERTIES CORRADE_USE_PEDANTIC_FLAGS ON)

configure_file(${CMAKE_CURRENT_SOURCE_DIR}/configure.h.cmake
               ${CMAKE_CURRENT_BINARY_DIR}/configure.h)

add_executable(magnum-multiwindow MultiWindowExample.cpp)
target_include_directories(magnum-multiwindow PRIVATE ${CMAKE_CURRENT_BINARY_DIR})
target_link_libraries(magnum-multiwindow
    Magnum::Application
    Magnum::Magnum
    Magnum::MeshTools
    Magnum::SceneGraph
    Magnum::Shaders)

install(TARGETS magnum-multiwindow DESTINATION ${MAGNUM_BINARY_INSTALL_DIR})
install(FILES README.md DESTINATION ${MAGNUM_DATA_INSTALL_DIR}/examples RENAME README-multiwindow.md)
